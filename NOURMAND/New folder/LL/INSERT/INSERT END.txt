struct Node {
    int data;
    Node* next;
};



Node* insertEnd(Node* start, int value) {
    Node* newNode = new Node();
    newNode->data = value;
    newNode->next = nullptr;
    if (start == nullptr) {
        return newNode;
    }

    Node* temp = start;
    while (temp->next != nullptr) {
        temp = temp->next;
    }
    temp->next = newNode;
    return start;
}
