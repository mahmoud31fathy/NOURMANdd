struct Node {
    int data;
    Node* next;
};



Node* deleteNode(Node* start, int value) {
    if (start == nullptr) {
        cout << "List is empty" << endl;
        return nullptr;
    }

    if (start->data == value) {
        Node* temp = start;
        start = start->next;
        delete temp;
        cout << "Node deleted" << endl;
        return start;
    }

    Node* temp = start;
    Node* prev = nullptr;
    while (temp != nullptr && temp->data != value) {
        prev = temp;
        temp = temp->next;
    }

    if (temp == nullptr) {
        cout << "Node with value " << value << " not found" << endl;
    }
    else {
        prev->next = temp->next;
        delete temp;
        cout << "Node deleted" << endl;
    }

    return start;
}
