const int MAX_SIZE = 100; 


void search(int arr[], int size, int value)
{
    for (int i = 0; i < size; i++)
    {
        if (arr[i] == value)
        {
            cout << "Element " << value << " found at position " << i << ".\n";
            return;
        }
    }
    cout << "Element " << value << " not found.\n";
}
