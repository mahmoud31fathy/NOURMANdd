const int MAX_SIZE = 100; 

void display(int arr[], int size)
{
    if (size == 0)
    {
        cout << "Array is empty.\n";
    }
    else
    {
        cout << "Array: ";
        for (int i = 0; i < size; i++)
        {
            cout << arr[i] << " ";
        }
        cout << "\n";
    }
}
